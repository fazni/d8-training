Static methods


    class Unicode {

      public static function strlen($text) {
        if (static::getStatus() == static::STATUS_MULTIBYTE) {
          return mb_strlen($text);
        }
        else {
          // Do not count UTF-8 continuation bytes.
          return strlen(preg_replace("/[\x80-\xBF]/", '', $text));
        }
      }

    }

Traits


    trait RandomGeneratorTrait {

      protected $randomGenerator;

      public function randomString($length = 8) {
        return $this->getRandomGenerator()->string($length);
      }

      protected function getRandomGenerator() {
        if (!is_object($this->randomGenerator)) {
          $this->randomGenerator = new Random();
        }
        return $this->randomGenerator;
      }

    }

 class EntityWrapper {

   /**
    * Passes unknown calls through to the wrapped Entity object.
    */
   public function __call($method, $args) {
     return call_user_func_array([$this->entity, $method], $args);
   }

 }

__get()


        class Wysiwyg {

          public function __get($property) {
            if ($this->editor->hasButton($property)) {
              return $this->editor->getButton($property);
            }
            throw new ButtonNotPresentException($property);
          }

        }

__set()


        class Configuration {

          protected array $settings = array();

          public function __set($property, $value) {
            // Validate any unknown properties and
            // store them in $settings.
            if ($this->validateSetting($property, $value)) {
              $this->settings[$property] = $value;
            }
            else {
              throw new NoSuchPropertyException($property);
            }
          }

        }

No Dependency Injection


class UserLoginForm extends FormBase {

  protected $userStorage;

  protected $renderer;

  public function __construct() {
    $this->userStorage = new UserStorageInterface();
    $this->renderer = new RendererInterface();
  }

}

Factory methods


class UserLoginForm extends FormBase {

  public function __construct(UserStorageInterface $user_storage, RendererInterface $renderer) {
    $this->userStorage = $user_storage;
    $this->renderer = $renderer;
  }

  public static function create(ContainerInterface $container) {
    return new static(
      $container->get('entity.manager')->getStorage('user'),
      $container->get('renderer')
    );
  }

}
